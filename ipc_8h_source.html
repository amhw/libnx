182 /**
183  * @brief Prepares the header of an IPC command structure.
184  * @param cmd IPC command structure.
185  * @param sizeof_raw Size in bytes of the raw data structure to embed inside the IPC request
186  * @return Pointer to the raw embedded data structure in the request, ready to be filled out.
187  */
188 static inline void* ipcPrepareHeader(IpcCommand* cmd, size_t sizeof_raw) {
189  u32* buf = (u32*)armGetTls();
190  size_t i;
191  *buf++ = 4 | (cmd->NumStaticIn << 16) | (cmd->NumSend << 20) | (cmd->NumRecv << 24) | (cmd->NumTransfer << 28);
192 
193  u32* fill_in_size_later = buf;
194 
195  if (cmd->NumStaticOut > 0) {
196  *buf = (cmd->NumStaticOut + 2) << 10;
197  }
198  else {
199  *buf = 0;
200  }
201 
202  if (cmd->SendPid || cmd->NumHandlesCopy > 0 || cmd->NumHandlesMove > 0) {
203  *buf++ |= 0x80000000;
204  *buf++ = (!!cmd->SendPid) | (cmd->NumHandlesCopy << 1) | (cmd->NumHandlesMove << 5);
205 
206  if (cmd->SendPid)
207  buf += 2;
208 
209  for (i=0; i<(cmd->NumHandlesCopy + cmd->NumHandlesMove); i++)
210  *buf++ = cmd->Handles[i];
211  }
212  else {
213  buf++;
214  }
215 
216  for (i=0; i<cmd->NumStaticIn; i++, buf+=2) {
217  IpcStaticSendDescriptor* desc = (IpcStaticSendDescriptor*) buf;
218 
219  uintptr_t ptr = (uintptr_t) cmd->Statics[i];
220  desc->Addr = ptr;
221  desc->Packed = cmd->Indices[i] | (cmd->StaticSizes[i] << 16) |
222  (((ptr >> 32) & 15) << 12) | (((ptr >> 36) & 15) << 6);
223  }
224 
225  for (i=0; i<(cmd->NumSend + cmd->NumRecv + cmd->NumTransfer); i++, buf+=3) {
226  IpcBufferDescriptor* desc = (IpcBufferDescriptor*) buf;
227  desc->Size = cmd->BufferSizes[i];
228 
229  uintptr_t ptr = (uintptr_t) cmd->Buffers[i];
230  desc->Addr = ptr;
231  desc->Packed = cmd->Flags[i] |
232  (((ptr >> 32) & 15) << 28) | ((ptr >> 36) << 2);
233  }
234 
235  u32 padding = ((16 - (((uintptr_t) buf) & 15)) & 15) / 4;
236  u32* raw = (u32*) (buf + padding);
237 
238  size_t raw_size = (sizeof_raw/4) + 4;
239  buf += raw_size;
240 
241  u16* buf_u16 = (u16*) buf;
242 
243  for (i=0; i<cmd->NumStaticOut; i++) {
244  size_t off = cmd->NumStaticIn + i;
245  size_t sz = (uintptr_t) cmd->StaticSizes[off];
246 
247  buf_u16[i] = (sz > 0xFFFF) ? 0 : sz;
248  }
249 
250  size_t u16s_size = ((2*cmd->NumStaticOut) + 3)/4;
251  buf += u16s_size;
252  raw_size += u16s_size;
253 
254  *fill_in_size_later |= raw_size;
255 
256  for (i=0; i<cmd->NumStaticOut; i++, buf+=2) {
257  IpcStaticRecvDescriptor* desc = (IpcStaticRecvDescriptor*) buf;
258  size_t off = cmd->NumStaticIn + i;
259 
260  uintptr_t ptr = (uintptr_t) cmd->Statics[off];
261  desc->Addr = ptr;
262  desc->Packed = (ptr >> 32) | (cmd->StaticSizes[off] << 16);
263  }
264 
265  return (void*) raw;
266 }

298 /**
299  * @brief Parse an IPC command response into an IPC parsed command structure.
300  * @param IPC parsed command structure to fill in.
301  * @return Result code.
302  */
303 static inline Result ipcParse(IpcParsedCommand* r) {
304  u32* buf = (u32*)armGetTls();
305  u32 ctrl0 = *buf++;
306  u32 ctrl1 = *buf++;
307  size_t i;
308 
309  r->HasPid = false;
310  r->RawSize = (ctrl1 & 0x1ff) * 4;
311  r->NumHandles = 0;
312 
313  if (ctrl1 & 0x80000000) {
314  u32 ctrl2 = *buf++;
315 
316  if (ctrl2 & 1) {
317  r->HasPid = true;
318  r->Pid = *buf++;
319  r->Pid |= ((u64)(*buf++)) << 32;
320  }
321 
322  size_t num_handles = ((ctrl2 >> 1) & 15) + ((ctrl2 >> 5) & 15);
323  buf += num_handles;
324 
325  if (num_handles > 8)
326  num_handles = 8;
327 
328  for (i=0; i<num_handles; i++)
329  r->Handles[i] = *(buf-num_handles+i);
330 
331  r->NumHandles = num_handles;
332  }
333 
334  size_t num_statics = (ctrl0 >> 16) & 15;
335  buf += num_statics*2;
336 
337  size_t num_bufs = ((ctrl0 >> 20) & 15) + ((ctrl0 >> 24) & 15) + (((ctrl0 >> 28) & 15));
338  r->Raw = (void*)(((uintptr_t)(buf + num_bufs*3) + 15) &~ 15);
339 
340  if (num_bufs > 4)
341  num_bufs = 4;
342 
343  for (i=0; i<num_bufs; i++, buf+=3) {
344  IpcBufferDescriptor* desc = (IpcBufferDescriptor*) buf;
345  u64 packed = (u64) desc->Packed;
346 
347  r->Buffers[i] = (void*) (desc->Addr | ((packed >> 28) << 32) | (((packed >> 2) & 15) << 36));
348  r->BufferSizes[i] = desc->Size;
349  // todo: Do we care about buffer type?
350  }
351 
352  r->NumBuffers = num_bufs;
353  return 0;
354 }
